\newcommand{\send}{\Rightarrow}
\newcommand{\sendto}{\rightarrow}
\algrenewcommand\textproc{\textrm}

\begin{algorithmic}
  \algrenewcommand\algorithmicprocedure{\textbf{class}}
  \Procedure{$\text{DBS\_Peer}:\text{STS\_Peer}$}{$\text{load\_balancer, splitters\_tracker, channel}$}
  
  \Function{\text{main}}{$\text{load\_balancer, splitters\_tracker, channel}$}
  \State $\text{connected\_splitter}=\text{STS\_Peer.main}(\text{load\_balancer, splitters\_tracker, channel})$
  \State $\text{join\_team}(\text{connected\_splitter})$
  \State $\vdots$
  \EndFunction
  
  \Function{$\text{join\_team}$}{$\text{splitter}$}
  \State $[``\mathtt{hello}"] \send \text{plitter}$
  \ForAll{$P \in [\text{list\_of\_peers}] \gets \text{splitter}$}
  \State $[``\mathtt{hello}"] \sendto P$
  \EndFor
  \EndFunction

  \Function{$\text{get\_chunk}()$}{}
  \State $[\text{message}] \gets X$
  \If{$\text{message}.length() < \text{MAX\_MESSAGE\_LENGTH}$}
  \State \Return $\text{control\_message}(\text{message}, X)$
  \Else
  \State \Return $\text{stream\_message}(\text{message}, X)$
  \EndIf
  \EndFunction
  
  \Function{$\text{control\_message}$}{$\text{message}, \text{sender}$}
  \If{$\text{message}[0] = ``\text{HELLO\_CODE}"$}
  \State $\text{add\_neighbor}(\text{sender})$
  \EndIf
  \State $\vdots$
  \EndFunction

  \Function{$\text{add\_neighbor}$}{$\text{sender}$}
  \State $\text{forward}[\text{Peer.me}].\text{append}(\text{sender})$
  \State $\vdots$
  %\State $\text{neighbor} = \text{sender}$
  %\State $\text{debt}[\text{sender}]=0$
  \EndFunction

  $\vdots$

  \EndProcedure
\end{algorithmic}
