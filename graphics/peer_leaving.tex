\newcommand{\send}{\Rightarrow}
\newcommand{\sendto}{\rightarrow}
\algrenewcommand\algorithmicprocedure{\textbf{task}}

\begin{algorithmic}

  \Procedure{$P_o$.Leave}{$T$}
  \Repeat
  \State $[\mathtt{goodbye}] \sendto S$
  \State $[\mathtt{goodbye}] \gets S$ \textbf{with} timeout
  \Until{\textbf{not} timeout}
  \State $[\mathtt{goodbye}] \sendto S$
  \ForAll{$P_k\in T^o$}
  \State $[\mathtt{goodbye}] \sendto P_k$
  \EndFor
  \State relay\_pending\_chunks()
  \EndProcedure
  \newline
  \Procedure{$P_k$.RemoveNeighbors}{}
  \While{True}
  \State $[\mathtt{goodbye}] \gets P_o$
  \State $T^k = T^k \setminus P_o$
  \EndWhile
  \EndProcedure
  
\end{algorithmic}
