\newcommand{\send}{\Rightarrow}
\newcommand{\sendto}{\rightarrow}
\algrenewcommand\algorithmicprocedure{\textbf{class}}
\algrenewcommand\textproc{\textrm}

\begin{algorithmic}
  \Procedure{$\text{Peer}$}{$\text{load\_balancer, player}$}

  \Function{$\text{get\_tracker}$}{$\text{URL\_channel, load\_balancer}$}
  \State $[\mathtt{URL\_channel}] \send \text{load\_balancer}$
  \State \Return $\text{splitters\_tracker} \gets \text{load\_balancer}$
  \EndFunction
  
  \Function{$\text{get\_list}$}{$\text{splitters\_tracker, URL\_channel}$}
  \State $[\mathtt{URL\_channel}] \send \text{splitters\_tracker}$
  \State \Return $[\{\mathrm{splitters}\}] \gets \mathrm{splitters\_tracker}$
  \EndFunction

  \Function{$\text{connect\_fastest\_splitter}$}{$\{\mathrm{splitters}\}$}
  \State $\mathrm{queue}=[]$
  \ForAll{$S \in \{\text{splitter}\}$}
  \State $\text{connect\_to}(S, \text{queue})$
  \EndFor 
  \State $[\mathtt{URL\_channel}] \send \text{splitters\_tracker}$
  \State \Return $\mathrm{queue}[0]$
  \EndFunction

  \algrenewcommand\algorithmicfunction{\textbf{thread}}

  \Function{$\text{connect\_to}$}{$\text{splitter}, \text{queue}$}
  \State $\text{connection} = \mathrm{connect}(\text{splitter})$
  \If{$\text{queue.is\_empty()}$}
  \State $\text{queue}.\text{put}(\text{connection})$
  \EndIf
  \EndFunction

  \algrenewcommand\algorithmicfunction{\textbf{function}}
  
  \Function{$\text{main}$}{$\text{load\_balancer, player}$}
  \State $\text{URL\_channel} \gets \text{player}$
  %\State $\text{URL\_channel} \get \text{request\_to\_user}(``\mathtt{URL~channel}'')$
  \State $\text{splitters\_tracker} = \text{get\_tracker}(\text{URL\_channel, load\_balancer})$
  \State $\{\text{splitters}\} = \text{get\_list}(\text{splitters\_tracker})$
  \State $\text{connection} = \text{find\_connection}(\{\text{splitters}\})$
  \State $\vdots$
  \EndFunction
  \EndProcedure

\end{algorithmic}
