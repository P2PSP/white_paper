\begin{algorithmic}

  \Procedure{$P^t_k$.ChunkFlooding}{}
  \While{True}
  \State $[x,\mathtt{chunk}] \gets S^t$ \textbf{or} $[x,{\cal P}_i,\mathtt{chunk}] \gets {\cal P}_m$
  \If{$[x,\mathtt{chunk}] \gets S^t$}
  \ForAll {$P^t_n \in T^t_k)$}
  \State $[x,P^t_k,\mathtt{chunk}] \sendto P^t_n$
  \EndFor
  \EndIf
  \If{$[x,P^t_i,\mathtt{chunk}] \gets P^t_m$}
  \If{$P^t_m \notin T^t_k$}
  \State $T^t_k = T^t_k \cup P^t_i$
  \EndIf
  \ForAll{$P^t_n \in T^t_k) \setminus P^t_m$}
  \If{$P^t_k \in G_{{\cal P}_n}[{\cal P}_i]$}
  \State $[x,P^t_i,\mathtt{chunk}] \sendto P^t_n$
  \EndIf
  \EndFor
  \EndIf
  \EndWhile
  \EndProcedure
  
\end{algorithmic}
